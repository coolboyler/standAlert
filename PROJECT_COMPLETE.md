### 🔧 核心算法

#### 连胜计算
```javascript
function updateStreak() {
  const today = new Date().toDateString()
  const lastStand = getLastStandDate()

  if (lastStand === today) return

  const yesterday = getYesterdayDate()
  if (lastStand === yesterday) {
    currentStreak++
  } else {
    currentStreak = 1
  }
}
```

#### 宠物状态衰减
```javascript
setInterval(() => {
  pet.happiness = Math.max(0, pet.happiness - 0.1)
  pet.hunger = Math.max(0, pet.hunger - 0.05)
  pet.energy = Math.max(0, pet.energy - 0.08)
}, 60000)
```

#### 成就检查
```javascript
function checkAchievements() {
  const checks = {
    'first_stand': () => totalStands >= 1,
    'streak_3': () => currentStreak >= 3,
    'century': () => totalStands >= 100,
    'early_bird': () => {
      const hour = new Date().getHours()
      return hour >= 5 && hour <= 7
    }
  }

  achievements.forEach(ach => {
    if (!ach.unlocked && checks[ach.id]()) {
      unlockAchievement(ach)
    }
  })
}
```
